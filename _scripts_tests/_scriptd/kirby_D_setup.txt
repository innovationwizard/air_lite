## 2. POSTGRESQL DATABASE SETUP

### Install PostgreSQL
```bash
# Ubuntu/Debian
sudo apt-get update
sudo apt-get install postgresql postgresql-contrib

# MacOS
brew install postgresql
brew services start postgresql

# Windows
# Download from https://www.postgresql.org/download/windows/
```

### Create Database and User
```sql
-- Connect as postgres user
sudo -u postgres psql

-- Create database
CREATE DATABASE ai_refill;

-- Create user with proper permissions
CREATE USER ai_refill_user WITH PASSWORD 'secure_password_2025';
GRANT ALL PRIVILEGES ON DATABASE ai_refill TO ai_refill_user;
GRANT CREATE ON SCHEMA public TO ai_refill_user;

-- Grant usage on schema
\c ai_refill
GRANT USAGE ON SCHEMA public TO ai_refill_user;
GRANT CREATE ON SCHEMA public TO ai_refill_user;

-- Exit
\q
```

### Verify Connection
```bash
psql -h localhost -U ai_refill_user -d ai_refill
```

## 6. AWS PRODUCTION DEPLOYMENT

### RDS PostgreSQL Setup
```python
# Production database config
db_config = {
    'host': 'ai-refill-prod.cluster-xyz.us-east-1.rds.amazonaws.com',
    'port': 5432,
    'database': 'ai_refill_prod',
    'user': 'ai_refill_user',
    'password': os.getenv('DB_PASSWORD')
}
```

### Docker Deployment
```dockerfile
# Dockerfile
FROM python:3.11-slim

WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    gcc \
    postgresql-client \
    && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY ai_refill_extractor.py .
COPY data/ ./data/

CMD ["python", "ai_refill_extractor.py"]
```

### Docker Compose
```yaml
# docker-compose.yml
version: '3.8'
services:
  postgres:
    image: postgres:15
    environment:
      POSTGRES_DB: ai_refill
      POSTGRES_USER: ai_refill_user
      POSTGRES_PASSWORD: secure_password_2025
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data

  ai_refill:
    build: .
    depends_on:
      - postgres
    environment:
      DB_HOST: postgres
      DB_PORT: 5432
      DB_NAME: ai_refill
      DB_USER: ai_refill_user
      DB_PASSWORD: secure_password_2025
    volumes:
      - ./data:/app/data
      - ./logs:/app/logs

volumes:
  postgres_data:
```

## 7. MONITORING & VALIDATION

### Data Quality Checks
```sql
-- Check extraction status
SELECT * FROM extraction_audit 
ORDER BY extraction_timestamp DESC LIMIT 1;

-- Review quality metrics
SELECT 
    table_name,
    total_records,
    valid_records,
    data_quality_score
FROM data_quality_log 
WHERE extraction_timestamp = (
    SELECT MAX(extraction_timestamp) FROM data_quality_log
);

-- Validate referential integrity
SELECT 'Sales without Products' as issue, COUNT(*) as count
FROM sales_transactions s 
LEFT JOIN products p ON s.product_id = p.product_id 
WHERE p.product_id IS NULL;
```

### Performance Monitoring
```bash
# Check database connections
SELECT count(*) as active_connections 
FROM pg_stat_activity 
WHERE state = 'active';

# Monitor table sizes
SELECT 
    schemaname,
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as size
FROM pg_tables 
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;
```

## 8. SECURITY BEST PRACTICES

### Database Security
```sql
-- Create read-only user for analytics
CREATE USER ai_refill_readonly WITH PASSWORD 'readonly_password';
GRANT CONNECT ON DATABASE ai_refill TO ai_refill_readonly;
GRANT USAGE ON SCHEMA public TO ai_refill_readonly;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO ai_refill_readonly;

-- Enable SSL connections
ALTER SYSTEM SET ssl = on;
```

### Application Security
```python
# Use environment variables
import os
from dotenv import load_dotenv

load_dotenv()

db_config = {
    'host': os.getenv('DB_HOST'),
    'port': int(os.getenv('DB_PORT', 5432)),
    'database': os.getenv('DB_NAME'),
    'user': os.getenv('DB_USER'),
    'password': os.getenv('DB_PASSWORD')
}
```

## 9. TROUBLESHOOTING

### Common Issues
```bash
# Connection refused
# Check PostgreSQL service status
sudo systemctl status postgresql

# Permission denied
# Verify database user permissions
GRANT ALL PRIVILEGES ON DATABASE ai_refill TO ai_refill_user;

# Memory issues with large files
# Increase available memory or use chunked processing
export PYTHONHASHSEED=0
ulimit -v 8388608  # 8GB memory limit
```

### Log Analysis
```bash
# Monitor extraction logs
tail -f ai_refill_extraction.log

# Check PostgreSQL logs
sudo tail -f /var/log/postgresql/postgresql-15-main.log
```
